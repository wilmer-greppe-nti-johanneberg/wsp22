post('/games/:id/update') do
    id = params[:id].to_i
    title = params[:title]
    price = params[:price].to_f
    tags = [
        params[:tag1],
        params[:tag2],
        params[:tag3],
        params[:tag4],
        params[:tag5]
    ]
    studio_name = params[:studio_name]
    update_game_info(id, title, price, tags, studio_name)
    redirect('/games')
end
